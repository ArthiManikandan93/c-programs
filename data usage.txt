#include<stdio.h>

int main() {
    int N;
    int data;
    int total = 0;
    int datausage = 0;

    scanf("%d", &N);

    for(int i = 0; i < N; i++) {
        scanf("%d", &data);

        total = total + data;

        if(data > 100) {
            datausage++;
        }
    }

    printf("Total Data: %d\n", total);
    printf("High Usage Days: %d", datausage);

    return 0;
}
