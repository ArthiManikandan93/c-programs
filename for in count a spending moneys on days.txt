// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int n;
    scanf("%d",&n);
    int expenses;
    int total=0;
    int overspending=0;
    
    for(int i =0;i<n;i++){
        scanf("%d",&expenses);
        total += expenses;
        if(expenses>1000){
            overspending++;
            
        }
    }
    printf("Total expenses:%d\n",total);
    printf("overspending Days:%d\n",overspending);

    return 0;
}
