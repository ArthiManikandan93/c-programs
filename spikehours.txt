#include <stdio.h>

int main() {
    int N;
    
    scanf("%d", &N);

    int unit;
    int totalunits = 0;
    int spikehour = 0;

    for(int i = 0; i < N; i++) {
        scanf("%d", &unit);
        
        totalunits = totalunits + unit;

        if(unit > 5) {
           spikehour++;
        }
    }

    printf("total units: %d\n",  totalunits);
    printf("spike hour: %d\n",spikehour);

    return 0;
}
