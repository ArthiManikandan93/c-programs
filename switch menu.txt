#include <stdio.h>

int main() {
    int ch,a,b;
    
    scanf("%d %d %d",&ch,&a,&b);
    switch(ch)
    {
        case 1:
            printf("Sum=%d",a+b);
            break;
        case 2:
            printf("Difference=%d",a-b);
            break;
        case 3:
            printf("Multiplication=%d",a*b);
            break;
        case 4:
          if(b!=0)
            printf("division=%d",a/b);
          else
            printf("division by zero");
            break;
        default:
           printf("Invalid choice");    
    }
    
    return 0;

}
