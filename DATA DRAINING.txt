#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int data;
    int maxdata = 0;
    int successfuldays = 0;

    for(int i = 0; i < N; i++) {
        scanf("%d", &data);

        if(i == 0) {
            maxdata = data;   
        }

        if(data > maxdata) {
            maxdata = data;   
        }

        if(data > 10) {
            successfuldays++;  
        }
    }

    printf("Maximum Data: %d\n", maxdata);
    printf("Successful Days: %d\n", successfuldays);

    return 0;
}
